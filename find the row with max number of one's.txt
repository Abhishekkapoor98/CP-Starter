int n,m;
int mx = INT_MIN;

int count_1(vvi mat)
{
	int index = -1;
	for(int i=0;i<n;i++)
	{
		int pos = lower_bound(mat[i].begin(),mat[i].end(),1) - mat[i].begin();
		int one = m - pos;
		if(one > mx)
		{
			mx = one;
			index = i;
		}
	}
	return index;
}
