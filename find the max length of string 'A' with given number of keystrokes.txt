int find_optimal(int n)
{
	if(n <= 6)
		return n;

	int mx = 0;

	for(int i = n-3; i >= 1;i--)
	{
		int curr = (n-i-1) * find_optimal(i);

		if(curr > mx)
			mx = curr;
	}

	return mx;
}
