int find_missing(vi arr,int low,int high,int diff)
{
	if(low < high)
	{
		int mid = (low+high)/2;

		if(arr[mid+1] - arr[mid] != diff)
			return (arr[mid]+diff);
		
		if(mid > 0 && arr[mid] - arr[mid-1] != diff)
			return arr[mid-1]+diff;

		if(arr[mid] == ((arr[0]+mid)*diff))
			return find_missing(arr,mid+1,high,diff);
		
		return find_missing(arr,low,mid,diff);
	}
	else
		return INT_MAX;
}

void solve()
{
	int tc;

		int n;

		vi arr(n);
		for(int i=0;i<n;i++)
			cin>>arr[i];

		int diff = (arr[n-1] - arr[0])/n;
		debug(diff);
		cout<<find_missing(arr,0,n-1,diff)<<endl;
	}
}
